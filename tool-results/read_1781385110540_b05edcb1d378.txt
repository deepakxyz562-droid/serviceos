   236→function formatCurrency(amount: number): string {
   237→  return new Intl.NumberFormat('en-US', {
   238→    style: 'currency', currency: 'USD',
   239→    minimumFractionDigits: 0, maximumFractionDigits: 0,
   240→  }).format(amount);
   241→}
   242→
   243→function formatDate(dateStr: string | null | undefined): string {
   244→  if (!dateStr) return '—';
   245→  try {
   246→    return new Date(dateStr).toLocaleDateString('en-US', {
   247→      month: 'short', day: 'numeric', year: 'numeric',
   248→    });
   249→  } catch { return dateStr; }
   250→}
   251→
   252→function formatDateTime(dateStr: string | null | undefined): string {
   253→  if (!dateStr) return '—';
   254→  try {
   255→    return new Date(dateStr).toLocaleString('en-US', {
   256→      month: 'short', day: 'numeric', year: 'numeric',
   257→      hour: '2-digit', minute: '2-digit',
   258→    });
   259→  } catch { return dateStr; }
   260→}

   352→  // Feature flags state

   356→  // Menu items state

   358→  const { data: menuData, isLoading: menuLoading } = useMenuItems(selectedTenantForMenu);
   359→
   360→  // Derived data
   361→  const tenants: Tenant[] = useMemo(() => {
   362→    if (!tenantsData) return [];

   367→  const subscriptions: Subscription[] = useMemo(() => {

   376→    return arr as UserRecord[];
   377→  }, [usersData]);
   378→
   379→  const featureFlags: FeatureFlagDef[] = useMemo(() => {
   380→    if (!flagsData) return FEATURE_DEFINITIONS.map((d) => ({ ...d, enabled: false }));
   381→    const arr = Array.isArray(flagsData) ? flagsData : (flagsData as Record<string, unknown>)?.flags || [];
   382→    // Merge with definitions to ensure all features appear
   383→    return FEATURE_DEFINITIONS.map((def) => {
   384→      const existing = (arr as FeatureFlagDef[]).find((f) => f.key === def.key);
   385→      return { ...def, enabled: existing?.enabled ?? false, config: existing?.config };
   386→    });
   387→  }, [flagsData]);
   388→
   389→  const menuItems: MenuItemDef[] = useMemo(() => {
   390→    if (!menuData) return DEFAULT_MENU_ITEMS.map((item, i) => ({ ...item, id: `default_${item.key}`, enabled: true, sortOrder: i }));
   391→    const arr = Array.isArray(menuData) ? menuData : (menuData as Record<string, unknown>)?.items || [];
   392→    return arr as MenuItemDef[];
   393→  }, [menuData]);
   394→
   395→  // Auto-select first tenant
   396→  useEffect(() => {
   397→    if (tenants.length > 0 && !selectedTenantForFlags) {
   398→      setSelectedTenantForFlags(tenants[0].id);

   403→    if (tenants.length > 0 && !selectedTenantForMenu) {

   408→  // ═══════════════════════════════════════════════════════════════════════════
   409→  // 1. DASHBOARD TAB
   410→  // ═══════════════════════════════════════════════════════════════════════════
   411→
   412→  function DashboardTab() {
   413→    const [stats, setStats] = useState<PlatformStats | null>(null);
   414→    const [loading, setLoading] = useState(true);
   415→
   416→    const fetchStats = useCallback(async () => {
   417→      setLoading(true);
   418→      try {
   419→        const res = await fetch('/api/superadmin/stats');
   420→        if (res.ok) {
   421→          const data = await res.json();
   422→          setStats(data);
   423→        } else {
   424→          setStats(getFallbackStats());
   425→        }
   426→      } catch {
   427→        setStats(getFallbackStats());
   428→      } finally {
   429→        setLoading(false);
   430→      }
   431→    }, []);
   432→
   433→    useEffect(() => { fetchStats(); }, [fetchStats]);
   434→
   435→    // Tenant growth chart data (from tenants by month)
   436→    const tenantGrowthData = useMemo(() => {
   437→      if (!tenants.length) return [];
   438→      const months: Record<string, number> = {};
   439→      const now = new Date();
   440→      for (let i = 5; i >= 0; i--) {
   441→        const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
   442→        const key = d.toLocaleDateString('en-US', { month: 'short' });
   443→        months[key] = 0;
   444→      }
   445→      tenants.forEach((t) => {
   446→        try {
   447→          const d = new Date(t.createdAt);
   448→          const key = d.toLocaleDateString('en-US', { month: 'short' });
   449→          if (key in months) months[key]++;
   450→        } catch { /* ignore */ }
   451→      });
   452→      return Object.entries(months).map(([month, count]) => ({ month, count }));
   453→    }, [tenants]);
   454→
   455→    const maxGrowth = Math.max(...tenantGrowthData.map((d) => d.count), 1);
   456→
   457→    // Recent signups (last 5 tenants)
   458→    const recentSignups = useMemo(() =>
   459→      [...tenants].sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime()).slice(0, 5),
   460→    [tenants]);
   461→
   462→    // Platform alerts
   463→    const alerts = useMemo(() => {
   464→      const list: { type: 'warning' | 'error' | 'info'; message: string; tenant?: string }[] = [];
   465→      tenants.filter((t) => t.planStatus === 'suspended').forEach((t) => {
   466→        list.push({ type: 'error', message: `Tenant "${t.name}" is suspended`, tenant: t.name });
   467→      });
   468→      tenants.filter((t) => t.planStatus === 'trial').forEach((t) => {
   469→        list.push({ type: 'warning', message: `Tenant "${t.name}" is on trial`, tenant: t.name });
   470→      });
   471→      if (list.length === 0) {
   472→        list.push({ type: 'info', message: 'No active alerts. Platform is healthy.' });
   473→      }
   474→      return list;
   475→    }, [tenants]);
   476→
   477→    if (loading) return <KPISkeleton count={5} />;
   478→
   479→    const kpiCards = [
   480→      { label: 'Total Tenants', value: stats?.totalTenants ?? 0, icon: Building2, trend: stats?.trends?.tenants, color: 'emerald' },
   481→      { label: 'Active Users', value: stats?.activeUsers ?? stats?.totalUsers ?? 0, icon: Users, trend: stats?.trends?.users, color: 'sky' },

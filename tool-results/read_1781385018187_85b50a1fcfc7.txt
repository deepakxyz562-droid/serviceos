   231→function formatUSD(amount: number): string {
   232→  return new Intl.NumberFormat('en-US', {
   233→    style: 'currency',
   234→    currency: 'USD',
   235→    minimumFractionDigits: 0,
   236→    maximumFractionDigits: 0,
   237→  }).format(amount);
   238→}

   240→function formatNumber(num: number): string {
   241→  return new Intl.NumberFormat('en-US').format(num);
   242→}

   244→function formatDate(dateStr: string | null): string {
   245→  if (!dateStr) return '—';
   246→  return new Date(dateStr).toLocaleDateString('en-US', {
   247→    day: '2-digit',
   248→    month: 'short',
   249→    year: 'numeric',
   250→  });
   251→}

   351→export function SuperAdminView() {
   352→  const [activeTab, setActiveTab] = useState('overview');
   353→  const [stats, setStats] = useState<PlatformStats | null>(null);
   354→  const [tenants, setTenants] = useState<TenantRecord[]>([]);
   355→  const [users, setUsers] = useState<UserRecord[]>([]);
   356→  const [billingData, setBillingData] = useState<BillingStats | null>(null);
   357→  const [whatsappData, setWhatsappData] = useState<WhatsAppStats | null>(null);
   358→  const [featureFlags, setFeatureFlags] = useState<FeatureFlagRecord[]>([]);
   359→  const [loading, setLoading] = useState(true);
   360→  const [error, setError] = useState<string | null>(null);
   361→
   362→  // Dialog states
   363→  const [createTenantOpen, setCreateTenantOpen] = useState(false);
   364→  const [usageDialogOpen, setUsageDialogOpen] = useState(false);
   365→  const [selectedTenant, setSelectedTenant] = useState<TenantRecord | null>(null);
   366→  const [newTenant, setNewTenant] = useState({ name: '', plan: 'starter', email: '', ownerName: '', password: '' });
   367→  const [creating, setCreating] = useState(false);
   368→
   369→  // Filter states
   370→  const [tenantSearch, setTenantSearch] = useState('');
   371→  const [tenantStatusFilter, setTenantStatusFilter] = useState('all');
   372→  const [userSearch, setUserSearch] = useState('');
   373→  const [subStatusFilter, setSubStatusFilter] = useState('all');
   374→  const [billingCycle, setBillingCycle] = useState<'monthly' | 'yearly'>('monthly');
   375→
   376→  // ─── Data Fetching ───────────────────────────────────────────────────────
   377→
   378→  const fetchAllData = useCallback(async () => {
   379→    setLoading(true);
   380→    setError(null);
   381→    try {
   382→      const endpoints = [
   383→        { url: '/api/admin/stats', key: 'stats' },
   384→        { url: '/api/admin/tenants', key: 'tenants' },
   385→        { url: '/api/admin/users', key: 'users' },
   386→        { url: '/api/admin/billing', key: 'billing' },
   387→        { url: '/api/admin/whatsapp', key: 'whatsapp' },
   388→        { url: '/api/admin/feature-flags', key: 'featureFlags' },
   389→      ];
   390→
   391→      const results = await Promise.allSettled(
   392→        endpoints.map(({ url }) =>
   393→          fetch(url).then((r) => {
   394→            if (!r.ok) throw new Error(`Failed to fetch ${url}`);
   395→            return r.json();
   396→          })
   397→        )
   398→      );
   399→
   400→      results.forEach((result, i) => {
   401→        if (result.status === 'fulfilled') {
   402→          const data = result.value;
   403→          const key = endpoints[i].key;
   404→          switch (key) {
   405→            case 'stats': setStats(data); break;
   406→            case 'tenants': setTenants(data.tenants || data || []); break;
   407→            case 'users': setUsers(data.users || data || []); break;
   408→            case 'billing': setBillingData(data); break;
   409→            case 'whatsapp': setWhatsappData(data); break;
   410→            case 'featureFlags': setFeatureFlags(data.featureFlags || data || []); break;
   411→          }
   412→        }
   413→      });
   414→    } catch (err) {
   415→      setError(err instanceof Error ? err.message : 'Failed to load admin data');
   416→    } finally {
   417→      setLoading(false);
   418→    }
   419→  }, []);
   420→
   421→  useEffect(() => {
   422→    fetchAllData();
   423→  }, [fetchAllData]);
   424→
   425→  // ─── Derived Data ───────────────────────────────────────────────────────
   426→
   427→  const subscriptions = useMemo(() => {
   428→    return tenants.map((t) => ({
   429→      id: t.id,
   430→      companyName: t.name,
   431→      plan: t.plan,
   432→      status: t.planStatus === 'suspended' ? 'suspended' as const : t.planStatus as 'active' | 'trial' | 'expired' | 'cancelled',
   433→      users: t.userCount,
   434→      jobs: t.jobCount,
   435→      renewalDate: t.trialEndsAt || t.createdAt,
   436→      mrr: planConfig[t.plan]?.monthlyPrice || 29,
   437→    }));
   438→  }, [tenants]);
   439→
   440→  const tenantFeatureMap = useMemo(() => {
   441→    const map: Record<string, FeatureFlagRecord['featureFlags']> = {};
   442→    featureFlags.forEach((f) => {
   443→      map[f.tenantId] = f.featureFlags;
   444→    });
   445→    return map;
   446→  }, [featureFlags]);
   447→
   448→  // ─── Filtered Data ──────────────────────────────────────────────────────
   449→
   450→  const filteredTenants = useMemo(() => {
   451→    return tenants.filter((t) => {
   452→      const matchesSearch = t.name.toLowerCase().includes(tenantSearch.toLowerCase()) ||
   453→        (t.email || '').toLowerCase().includes(tenantSearch.toLowerCase());
   454→      const matchesStatus = tenantStatusFilter === 'all' || t.planStatus === tenantStatusFilter ||
   455→        (tenantStatusFilter === 'suspended' && t.suspendedAt);
   456→      return matchesSearch && matchesStatus;
   457→    });
   458→  }, [tenants, tenantSearch, tenantStatusFilter]);
   459→
   460→  const filteredSubscriptions = useMemo(() => {
   461→    return subscriptions.filter((s) => {
   462→      return subStatusFilter === 'all' || s.status === subStatusFilter;
   463→    });
   464→  }, [subscriptions, subStatusFilter]);
   465→
   466→  const filteredUsers = useMemo(() => {
   467→    return users.filter((u) => {
   468→      const q = userSearch.toLowerCase();
   469→      return (
   470→        u.name.toLowerCase().includes(q) ||
   471→        u.email.toLowerCase().includes(q) ||
   472→        (u.tenantName || '').toLowerCase().includes(q)
   473→      );
   474→    });
   475→  }, [users, userSearch]);
   476→
   477→  // ─── Revenue Chart Data ─────────────────────────────────────────────────
   478→
   479→  const revenueChartData = useMemo(() => {
   480→    if (stats?.monthlyRevenueData && stats.monthlyRevenueData.length > 0) {
   481→      return stats.monthlyRevenueData.map((d) => ({
   482→        month: d.label || d.month,
   483→        revenue: d.revenue,
   484→      }));
   485→    }
   486→    // Fallback from stats
   487→    const months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
   488→    const baseMrr = stats?.mrr || 5000;
   489→    return months.map((m, i) => ({
   490→      month: m,
   491→      revenue: Math.round(baseMrr * (0.4 + i * 0.06)),
   492→    }));
   493→  }, [stats]);
   494→
   495→  // ─── Analytics Data ─────────────────────────────────────────────────────
   496→
   497→  const analyticsData = useMemo(() => {
   498→    if (!stats) return null;
   499→    const months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
   500→    const base = Math.max(stats.totalTenants, 1);
   501→    return {
   502→      customersOverTime: months.map((m, i) => ({
   503→        month: m,
   504→        count: Math.round(base * (0.4 + i * 0.06)),
   505→      })),
   506→      leadsOverTime: months.map((m, i) => ({
   507→        month: m,
   508→        count: Math.round(stats.totalLeads * (0.3 + i * 0.07)),
   509→      })),
   510→      jobsOverTime: months.map((m, i) => ({
   511→        month: m,
   512→        count: Math.round(stats.totalJobs * (0.3 + i * 0.06)),
   513→      })),
   514→      mrrArr: months.map((m, i) => ({
   515→        month: m,
   516→        mrr: Math.round(stats.mrr * (0.5 + i * 0.05)),
   517→        arr: Math.round(stats.arr * (0.5 + i * 0.05)),
   518→      })),
   519→      churnRate: stats.churnRate,
   520→      trialConversionRate: stats.trialConversionRate,
   521→    };
   522→  }, [stats]);
   523→
   524→  // ─── Action Handlers ────────────────────────────────────────────────────
   525→
   526→  const handleCreateTenant = async () => {
   527→    if (!newTenant.name.trim() || !newTenant.email.trim() || !newTenant.ownerName.trim() || !newTenant.password.trim()) {
   528→      toast.error('Please fill in all required fields');
   529→      return;
   530→    }
   531→    setCreating(true);
   532→    try {
   533→      const res = await authFetch('/api/admin/tenants', {
   534→        method: 'POST',
   535→        headers: { 'Content-Type': 'application/json' },
   536→        body: JSON.stringify({
   537→          businessName: newTenant.name,
   538→          ownerName: newTenant.ownerName,
   539→          ownerEmail: newTenant.email,
   540→          password: newTenant.password,
   541→          plan: newTenant.plan,
   542→        }),
   543→      });
   544→      if (!res.ok) {
   545→        const errData = await res.json();
   546→        throw new Error(errData.error || 'Failed to create tenant');
   547→      }
   548→      const created = await res.json();
   549→      setCreateTenantOpen(false);
   550→      setNewTenant({ name: '', plan: 'starter', email: '', ownerName: '', password: '' });
   551→      toast.success(`Company "${created.tenant?.name || newTenant.name}" created successfully`);
   552→      fetchAllData(); // Refresh all data
   553→    } catch (err) {
   554→      toast.error(err instanceof Error ? err.message : 'Failed to create company');
   555→    } finally {
   556→      setCreating(false);
   557→    }
   558→  };
   559→
   560→  const handleTenantAction = async (tenantId: string, action: 'suspend' | 'activate' | 'delete') => {
   561→    try {
   562→      const res = await authFetch('/api/admin/tenants', {
   563→        method: 'PUT',
   564→        headers: { 'Content-Type': 'application/json' },
   565→        body: JSON.stringify({ id: tenantId, action }),
   566→      });
   567→      if (!res.ok) throw new Error(`Failed to ${action} tenant`);
   568→
   569→      if (action === 'delete') {
   570→        setTenants((prev) => prev.filter((t) => t.id !== tenantId));
   571→        toast.success('Company deleted successfully');
   572→      } else if (action === 'suspend') {
   573→        setTenants((prev) =>
   574→          prev.map((t) => (t.id === tenantId ? { ...t, planStatus: 'suspended', suspendedAt: new Date().toISOString() } : t))
   575→        );
   576→        toast.success('Company suspended');
   577→      } else {
   578→        setTenants((prev) =>

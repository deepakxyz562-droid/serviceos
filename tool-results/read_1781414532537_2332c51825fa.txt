   245→function formatDate(dateStr: string | null | undefined): string {
   246→  if (!dateStr) return '—';
   247→  try {
   248→    return new Date(dateStr).toLocaleDateString('en-US', {
   249→      month: 'short', day: 'numeric', year: 'numeric',
   250→    });
   251→  } catch { return dateStr; }
   252→}
   253→
   254→function formatDateTime(dateStr: string | null | undefined): string {
   255→  if (!dateStr) return '—';
   256→  try {
   257→    return new Date(dateStr).toLocaleString('en-US', {
   258→      month: 'short', day: 'numeric', year: 'numeric',
   259→      hour: '2-digit', minute: '2-digit',
   260→    });
   261→  } catch { return dateStr; }
   262→}

   423→  function DashboardTab() {
   424→    const [stats, setStats] = useState<PlatformStats | null>(null);
   425→    const [loading, setLoading] = useState(true);
   426→
   427→    const fetchStats = useCallback(async () => {
   428→      setLoading(true);
   429→      try {
   430→        const res = await fetch('/api/superadmin/stats');
   431→        if (res.ok) {
   432→          const data = await res.json();
   433→          setStats(data);
   434→        } else {
   435→          setStats(getFallbackStats());
   436→        }
   437→      } catch {
   438→        setStats(getFallbackStats());
   439→      } finally {
   440→        setLoading(false);
   441→      }
   442→    }, []);
   443→
   444→    useEffect(() => { fetchStats(); }, [fetchStats]);
   445→
   446→    // Tenant growth chart data (from tenants by month)
   447→    const tenantGrowthData = useMemo(() => {
   448→      if (!tenants.length) return [];
   449→      const months: Record<string, number> = {};
   450→      const now = new Date();
   451→      for (let i = 5; i >= 0; i--) {
   452→        const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
   453→        const key = d.toLocaleDateString('en-US', { month: 'short' });
   454→        months[key] = 0;
   455→      }
   456→      tenants.forEach((t) => {
   457→        try {
   458→          const d = new Date(t.createdAt);
   459→          const key = d.toLocaleDateString('en-US', { month: 'short' });
   460→          if (key in months) months[key]++;
   461→        } catch { /* ignore */ }
   462→      });
   463→      return Object.entries(months).map(([month, count]) => ({ month, count }));
   464→    }, [tenants]);
   465→
   466→    const maxGrowth = Math.max(...tenantGrowthData.map((d) => d.count), 1);
   467→
   468→    // Recent signups (last 5 tenants)
   469→    const recentSignups = useMemo(() =>
   470→      [...tenants].sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime()).slice(0, 5),
   471→    [tenants]);
   472→
   473→    // Platform alerts
   474→    const alerts = useMemo(() => {
   475→      const list: { type: 'warning' | 'error' | 'info'; message: string; tenant?: string }[] = [];
   476→      tenants.filter((t) => t.planStatus === 'suspended').forEach((t) => {
   477→        list.push({ type: 'error', message: `Tenant "${t.name}" is suspended`, tenant: t.name });
   478→      });
   479→      tenants.filter((t) => t.planStatus === 'trial').forEach((t) => {
   480→        list.push({ type: 'warning', message: `Tenant "${t.name}" is on trial`, tenant: t.name });

  1264→  function FeatureFlagsTab() {
  1265→    const [localFlags, setLocalFlags] = useState<FeatureFlagDef[]>([]);
  1266→    const [saving, setSaving] = useState(false);
  1267→
  1268→    useEffect(() => {
  1269→      setLocalFlags(featureFlags);
  1270→    }, [featureFlags]);
  1271→
  1272→    const handleToggle = (key: string) => {
  1273→      const flag = localFlags.find((f) => f.key === key);
  1274→      if (!flag || !selectedTenantForFlags) return;
  1275→
  1276→      const newEnabled = !flag.enabled;
  1277→      setLocalFlags((prev) => prev.map((f) => f.key === key ? { ...f, enabled: newEnabled } : f));
  1278→
  1279→      toggleFeatureFlagMutation.mutate(
  1280→        { tenantId: selectedTenantForFlags, featureKey: key, enabled: newEnabled },
  1281→        {
  1282→          onError: () => {
  1283→            setLocalFlags((prev) => prev.map((f) => f.key === key ? { ...f, enabled: !newEnabled } : f));
  1284→            toast.error('Failed to toggle feature flag');
  1285→          },
  1286→          onSuccess: () => {
  1287→            toast.success(`${flag.label} ${newEnabled ? 'enabled' : 'disabled'}`);
  1288→          },
  1289→        },
  1290→      );
  1291→    };
  1292→
  1293→    const handleEnableAll = () => {
  1294→      if (!selectedTenantForFlags) return;
  1295→      setLocalFlags((prev) => prev.map((f) => ({ ...f, enabled: true })));
  1296→      // Save all
  1297→      setSaving(true);
  1298→      fetch('/api/superadmin/feature-flags', {
  1299→        method: 'POST',
  1300→        headers: { 'Content-Type': 'application/json' },
  1301→        body: JSON.stringify({ tenantId: selectedTenantForFlags, flags: localFlags.map((f) => ({ key: f.key, enabled: true })) }),
  1302→      }).then(() => { toast.success('All features enabled'); setSaving(false); }).catch(() => { toast.error('Failed'); setSaving(false); });
  1303→    };
  1304→
  1305→    const handleDisableAll = () => {
  1306→      if (!selectedTenantForFlags) return;
  1307→      setLocalFlags((prev) => prev.map((f) => ({ ...f, enabled: false })));
  1308→      setSaving(true);
  1309→      fetch('/api/superadmin/feature-flags', {
  1310→        method: 'POST',
  1311→        headers: { 'Content-Type': 'application/json' },

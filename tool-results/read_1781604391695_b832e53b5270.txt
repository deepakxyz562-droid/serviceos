   252→function formatDate(dateStr: string | null | undefined): string {
   253→  if (!dateStr) return '—';
   254→  try {
   255→    return new Date(dateStr).toLocaleDateString('en-US', {
   256→      month: 'short', day: 'numeric', year: 'numeric',
   257→    });
   258→  } catch { return dateStr; }
   259→}
   260→
   261→function formatDateTime(dateStr: string | null | undefined): string {
   262→  if (!dateStr) return '—';
   263→  try {
   264→    return new Date(dateStr).toLocaleString('en-US', {
   265→      month: 'short', day: 'numeric', year: 'numeric',
   266→      hour: '2-digit', minute: '2-digit',
   267→    });
   268→  } catch { return dateStr; }
   269→}

   446→  function DashboardTab() {
   447→    // Use statsData from the parent hook (useSaasStats → /api/superadmin/stats)
   448→    const stats = (statsData as PlatformStats) || null;
   449→    const loading = statsLoading;
   450→
   451→    const handleRefresh = useCallback(() => {
   452→      refetchStats();
   453→      refetchTenants();
   454→    }, [refetchStats, refetchTenants]);
   455→
   456→    // Tenant growth chart data (from tenants by month)
   457→    const tenantGrowthData = useMemo(() => {
   458→      if (!tenants.length) return [];
   459→      const months: Record<string, number> = {};
   460→      const now = new Date();
   461→      for (let i = 5; i >= 0; i--) {
   462→        const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
   463→        const key = d.toLocaleDateString('en-US', { month: 'short' });
   464→        months[key] = 0;
   465→      }
   466→      tenants.forEach((t) => {
   467→        try {

  1294→            toast.error('Failed to toggle feature flag');

  1297→            toast.success(`${flag.label} ${newEnabled ? 'enabled' : 'disabled'}`);

  1300→      );

  1302→
  1303→    const handleEnableAll = () => {

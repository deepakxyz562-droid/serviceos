   244→function formatDate(dateStr: string | null | undefined): string {
   245→  if (!dateStr) return '—';
   246→  try {
   247→    return new Date(dateStr).toLocaleDateString('en-US', {
   248→      month: 'short', day: 'numeric', year: 'numeric',
   249→    });
   250→  } catch { return dateStr; }
   251→}
   252→
   253→function formatDateTime(dateStr: string | null | undefined): string {
   254→  if (!dateStr) return '—';
   255→  try {
   256→    return new Date(dateStr).toLocaleString('en-US', {
   257→      month: 'short', day: 'numeric', year: 'numeric',
   258→      hour: '2-digit', minute: '2-digit',
   259→    });
   260→  } catch { return dateStr; }
   261→}

   417→  // ═══════════════════════════════════════════════════════════════════════════
   418→  // 1. DASHBOARD TAB

   421→  function DashboardTab() {
   422→    const [stats, setStats] = useState<PlatformStats | null>(null);
   423→    const [loading, setLoading] = useState(true);
   424→
   425→    const fetchStats = useCallback(async () => {
   426→      setLoading(true);
   427→      try {
   428→        const res = await fetch('/api/superadmin/stats');
   429→        if (res.ok) {
   430→          const data = await res.json();
   431→          setStats(data);
   432→        } else {
   433→          setStats(getFallbackStats());
   434→        }
   435→      } catch {
   436→        setStats(getFallbackStats());
   437→      } finally {
   438→        setLoading(false);
   439→      }
   440→    }, []);
   441→
   442→    useEffect(() => { fetchStats(); }, [fetchStats]);
   443→
   444→    // Tenant growth chart data (from tenants by month)
   445→    const tenantGrowthData = useMemo(() => {
   446→      if (!tenants.length) return [];
   447→      const months: Record<string, number> = {};
   448→      const now = new Date();
   449→      for (let i = 5; i >= 0; i--) {
   450→        const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
   451→        const key = d.toLocaleDateString('en-US', { month: 'short' });
   452→        months[key] = 0;
   453→      }
   454→      tenants.forEach((t) => {
   455→        try {
   456→          const d = new Date(t.createdAt);
   457→          const key = d.toLocaleDateString('en-US', { month: 'short' });
   458→          if (key in months) months[key]++;
   459→        } catch { /* ignore */ }
   460→      });
   461→      return Object.entries(months).map(([month, count]) => ({ month, count }));
   462→    }, [tenants]);
   463→
   464→    const maxGrowth = Math.max(...tenantGrowthData.map((d) => d.count), 1);
   465→
   466→    // Recent signups (last 5 tenants)
   467→    const recentSignups = useMemo(() =>
   468→      [...tenants].sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime()).slice(0, 5),
   469→    [tenants]);
   470→
   471→    // Platform alerts
   472→    const alerts = useMemo(() => {
   473→      const list: { type: 'warning' | 'error' | 'info'; message: string; tenant?: string }[] = [];
   474→      tenants.filter((t) => t.planStatus === 'suspended').forEach((t) => {
   475→        list.push({ type: 'error', message: `Tenant "${t.name}" is suspended`, tenant: t.name });
   476→      });
   477→      tenants.filter((t) => t.planStatus === 'trial').forEach((t) => {
   478→        list.push({ type: 'warning', message: `Tenant "${t.name}" is on trial`, tenant: t.name });
   479→      });
   480→      if (list.length === 0) {
   481→        list.push({ type: 'info', message: 'No active alerts. Platform is healthy.' });
   482→      }

  1262→  function FeatureFlagsTab() {
  1263→    const [localFlags, setLocalFlags] = useState<FeatureFlagDef[]>([]);
  1264→    const [saving, setSaving] = useState(false);
  1265→
  1266→    useEffect(() => {
  1267→      setLocalFlags(featureFlags);
  1268→    }, [featureFlags]);
  1269→
  1270→    const handleToggle = (key: string) => {
  1271→      const flag = localFlags.find((f) => f.key === key);
  1272→      if (!flag || !selectedTenantForFlags) return;
  1273→
  1274→      const newEnabled = !flag.enabled;
  1275→      setLocalFlags((prev) => prev.map((f) => f.key === key ? { ...f, enabled: newEnabled } : f));
  1276→
  1277→      toggleFeatureFlagMutation.mutate(
  1278→        { tenantId: selectedTenantForFlags, featureKey: key, enabled: newEnabled },
  1279→        {
  1280→          onError: () => {
  1281→            setLocalFlags((prev) => prev.map((f) => f.key === key ? { ...f, enabled: !newEnabled } : f));
  1282→            toast.error('Failed to toggle feature flag');
  1283→          },
  1284→          onSuccess: () => {
  1285→            toast.success(`${flag.label} ${newEnabled ? 'enabled' : 'disabled'}`);
  1286→          },
  1287→        },
  1288→      );
  1289→    };
  1290→
  1291→    const handleEnableAll = () => {
  1292→      if (!selectedTenantForFlags) return;
  1293→      setLocalFlags((prev) => prev.map((f) => ({ ...f, enabled: true })));
  1294→      // Save all
  1295→      setSaving(true);
  1296→      fetch('/api/superadmin/feature-flags', {
  1297→        method: 'POST',
  1298→        headers: { 'Content-Type': 'application/json' },
  1299→        body: JSON.stringify({ tenantId: selectedTenantForFlags, flags: localFlags.map((f) => ({ key: f.key, enabled: true })) }),
  1300→      }).then(() => { toast.success('All features enabled'); setSaving(false); }).catch(() => { toast.error('Failed'); setSaving(false); });
  1301→    };
  1302→
  1303→    const handleDisableAll = () => {
  1304→      if (!selectedTenantForFlags) return;
  1305→      setLocalFlags((prev) => prev.map((f) => ({ ...f, enabled: false })));
  1306→      setSaving(true);
  1307→      fetch('/api/superadmin/feature-flags', {
  1308→        method: 'POST',
  1309→        headers: { 'Content-Type': 'application/json' },
  1310→        body: JSON.stringify({ tenantId: selectedTenantForFlags, flags: localFlags.map((f) => ({ key: f.key, enabled: false })) }),
  1311→      }).then(() => { toast.success('All features disabled'); setSaving(false); }).catch(() => { toast.error('Failed'); setSaving(false); });
  1312→    };
  1313→
  1314→    const enabledCount = localFlags.filter((f) => f.enabled).length;
